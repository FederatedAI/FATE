[docs]    def load_data(self, data_instance):
        # LOGGER.debug("In load_data of gradient function")
        X = []
        Y = []
        # 获取batch数据
        for iter_key, instant in data_instance:
            weighted_feature = instant.weight * instant.features
            X.append(weighted_feature)
            if instant.label == 1:
                Y.append([1])
            else:
                Y.append([-1])
        X = np.array(X)
        Y = np.array(Y)
        return X, Y
